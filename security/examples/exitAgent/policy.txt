// JADE code is allowed all permissions.
grant codebase "file:../../lib/coresec.jar" {
      permission java.security.AllPermission;  };
grant codebase "file:../../../../lib/jade.jar" {
      permission java.security.AllPermission;  };
grant codebase "file:../../../../lib/jadeTools.jar" {
      permission java.security.AllPermission;  };

// Agent code is not allowed any permission.
grant codebase "file:../../lib/examples.jar" {
};



grant principal jade.security.Name "*" {
  permission jade.security.PlatformPermission "", "create";
  permission jade.security.ContainerPermission "", "create";
  permission jade.security.AgentPermission "*", "create,kill";
  permission jade.security.AMSPermission   "agent-class=*", "register,deregister,modify";
  permission jade.security.MessagePermission "agent-name=*", "send-to";
};
